<script>
  const API_BASE = "http://localhost:5000";

  function usePrompt(el) {
    document.getElementById("promptInput").value = el.textContent.trim();
    document.getElementById("promptInput").focus();
  }

  function handleKey(e) {
    if (e.key === "Enter" && (e.ctrlKey || e.metaKey)) runQuery();
  }

  async function checkHealth() {
    try {
      const r = await fetch(`${API_BASE}/api/health`);
      const d = await r.json();
      const dot  = document.getElementById("statusDot");
      const text = document.getElementById("statusText");
      if (d.status === "ok") {
        dot.style.background = "#10b981";
        text.textContent = "DB Connected";
      } else {
        dot.style.background = "#ef4444";
        text.textContent = "DB Unreachable";
      }
    } catch {
      document.getElementById("statusDot").style.background = "#ef4444";
      document.getElementById("statusText").textContent = "Server Offline";
    }
  }

  async function runQuery() {
    const prompt = document.getElementById("promptInput").value.trim();
    if (!prompt) return;

    const btn  = document.getElementById("runBtn");
    const icon = document.getElementById("btnIcon");
    const txt  = document.getElementById("btnText");
    const out  = document.getElementById("results");

    btn.disabled = true;
    icon.textContent = "⏳";
    txt.textContent  = "Running…";

    out.style.display = "block";
    out.innerHTML = `
      <div class="loading-card">
        <div class="spinner"></div>
        <p>AI is writing SQL for: <span>"${escHtml(prompt)}"</span></p>
      </div>`;

    try {
      const res = await fetch(`${API_BASE}/api/query`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ prompt })
      });
      const data = await res.json();

      if (!data.success) {
        out.innerHTML = `
          <div class="error-card">
            <strong>❌ Error</strong>
            ${escHtml(data.error || "Unknown error occurred.")}
          </div>`;
        return;
      }

      renderResults(data);

    } catch (err) {
      out.innerHTML = `
        <div class="error-card">
          <strong>❌ Cannot reach Flask server</strong>
          Make sure the Python backend is running on port 5000.<br/>
          <code style="font-family:monospace;font-size:12px;color:#f87171">python -m backend.app</code>
        </div>`;
    } finally {
      btn.disabled = false;
      icon.textContent = "▶";
      txt.textContent  = "Run Query";
    }
  }

  function renderResults(data) {
    const out = document.getElementById("results");

    let tableHtml = "";
    if (data.rows && data.rows.length > 0) {
      const headers = data.columns.map(c =>
        `<th>${escHtml(c)}</th>`).join("");
      const rows = data.rows.map(row => {
        const cells = data.columns.map(col => {
          const val = row[col];
          return `<td>${formatCell(col, val)}</td>`;
        }).join("");
        return `<tr>${cells}</tr>`;
      }).join("");
      tableHtml = `
        <div class="table-wrap">
          <table>
            <thead><tr>${headers}</tr></thead>
            <tbody>${rows}</tbody>
          </table>
        </div>`;
    } else {
      tableHtml = `<div class="empty-cell">No records found for this query.</div>`;
    }

    out.innerHTML = `
      <div class="result-card">
        <div class="result-header">
          <div class="result-meta">
            <h3>Query Results</h3>
            <p>${escHtml(data.explanation || "")}</p>
          </div>
          <div class="row-badge">${data.row_count} row${data.row_count !== 1 ? "s" : ""}</div>
        </div>
        <div class="sql-block">
          <div class="sql-block-label">Generated SQL</div>
          <div class="sql-code">${escHtml(data.sql)}</div>
        </div>
        ${tableHtml}
      </div>`;
  }

  function formatCell(col, val) {
    if (val === null || val === undefined) return '<span style="color:#475569">NULL</span>';
    const str = String(val);
    const lc  = str.toLowerCase();
    const colL = col.toLowerCase();

    // Status/severity badges
    if (colL.includes("status") || colL === "severity") {
      const cls = {
        success:"success", failed:"failed", pending:"pending",
        critical:"critical", high:"high", medium:"medium", low:"low",
        active:"active", frozen:"frozen", closed:"closed"
      }[lc] || "";
      if (cls) return `<span class="badge badge-${cls}">${escHtml(str)}</span>`;
    }
    // Currency
    if (colL.includes("amount") || colL.includes("balance")) {
      const n = parseFloat(str);
      if (!isNaN(n)) return `₹${n.toLocaleString("en-IN", {minimumFractionDigits:2})}`;
    }
    // Datetime — make it readable
    if (colL.includes("_at") || colL.includes("date")) {
      return `<span style="color:#94a3b8">${escHtml(str)}</span>`;
    }
    return escHtml(str);
  }

  function escHtml(s) {
    return String(s)
      .replace(/&/g,"&amp;")
      .replace(/</g,"&lt;")
      .replace(/>/g,"&gt;")
      .replace(/"/g,"&quot;");
  }

  // Check health on load
  window.addEventListener("load", checkHealth);
</script>
